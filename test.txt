name: Restore Staging Branch

on:
  pull_request:
    types: [closed]
    branches:
      - main

jobs:
  restore-staging:
    if: github.event.pull_request.merged == true && github.event.pull_request.base.ref == 'main' && github.event.pull_request.head.ref == 'staging'
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v2
        with:
          ref: main

      - name: Set up Git
        run: |
          git config --global user.name 'github-actions[bot]'
          git config --global user.email 'github-actions[bot]@users.noreply.github.com'

      - name: Delete staging branch locally
        run: git branch -D staging || true

      - name: Create new staging branch from main
        run: git checkout -b staging

      - name: Push new staging branch to remote
        run: git push origin staging --force
